SELECT year, SUM(number) AS total_births
FROM some_teradata_db.some_teradata_schema.legacy_table
GROUP BY year
ORDER BY year;
SELECT year, name, MAX(number) AS max_number FROM some_teradata_db.some_teradata_schema.legacy_table GROUP BY year, name QUALIFY ROW_NUMBER() OVER (PARTITION BY year ORDER BY max_number DESC) = 1 ORDER BY year;

SELECT name, SUM(number) AS total_number
FROM some_teradata_db.some_teradata_schema.legacy_table
GROUP BY name
ORDER BY total_number DESC
FETCH FIRST 10 ROWS ONLY
;
